def iqr_method(data,var):
    q1 = np.percentile(data[var],25)
    q3= np.percentile(data[var],75)
    iqr = q3-q1
    lower_whisker = q1-1.5*iqr
    upper_whisker = q3+1.5*iqr
    return lower_whisker, upper_whisker

90% interval estimate for quantity variable¶

In [104]:
lst = df['Quantity (lts.)'].values.tolist()
conf_interval = 0.9
sample = len(lst)
std_error = np.std(lst)
moe = stats.t.ppf((1+conf_interval)/2, df = sample-1)*std_error
interval_estimate = (np.mean(lst) - moe,
                    np.mean(lst)+moe)
print (interval_estimate)
